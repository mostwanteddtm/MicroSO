	EXTERN DEBUGTINY:PROC
	EXTERN KEYPRESSED:BYTE
	EXTERN FULLBUFFER:BYTE
	PUSH CX

	@@:
		MOV		BYTE PTR FULLBUFFER, 1						; IF FULLBUFFER = 0 THEN CX DEFINE NOW MANY BYTES WILL BE COPIED
		CALL	DEBUGTINY
		XOR		AX, AX
		INT		16h
		
		CMP		AH, 01h
		JE		@f
		CMP		AH, 1Ch
		JE		@f
		
		MOV		KEYPRESSED, AH
		JMP		@b
	@@:

	POP 	CX
